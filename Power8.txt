import java.util.*;

public class Solution {
    public static void main(String[] args) {
        Scanner sc = new Scanner(System.in);
        int n = sc.nextInt();

        boolean isPower = true;

        if (n <= 0) {
            isPower = false;
        } else {
            while (n > 1) {
                if (n % 2 != 0) {
                    isPower = false;
                    break;
                }
                n = n / 2;
            }
        }

        if (isPower)
            System.out.println("Number is a power of two");
        else
            System.out.println("Number is not a power of two");
    }
}
